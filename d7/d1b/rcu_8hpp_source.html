32 // Hazard pointer implementation. Pointers form a linked list. \p ptr points
33 // to the data they 'hold', next - to the next element in a list.
34 // kUsed is a filler value to show that hazard pointer is not free. Please see
35 // comment to it. Please note, pointers do not form a linked list with
36 // application-wide list, that is there is no 'static
37 // std::atomic<HazardPointerRecord*> global_head' Every rcu::Variable has its
38 // own list of hazard pointers. Thus, move-assignment on hazard pointers is
39 // difficult to implement.
40 template <typename T>
41 struct HazardPointerRecord final {
42  // You see, objects are created 'filled', that is for the purposes of hazard
43  // pointer list, they contain value. This eliminates some race conditions,
44  // because these algorithms checks for ptr != nullptr (And kUsed is not
45  // nullptr). Obviously, this value can't be used, when dereferenced it points
46  // somewhere into kernel space and will cause SEGFAULT
47  static inline T* const kUsed = reinterpret_cast<T*>(1);
48 
49  explicit HazardPointerRecord(const Variable<T>& owner) : owner(owner) {}
50 
51  std::atomic<T*> ptr = kUsed;
52  const Variable<T>& owner;
53  std::atomic<HazardPointerRecord*> next{nullptr};
54 
55  // Simple operation that marks this hazard pointer as no longer used.
56  void Release() { ptr = nullptr; }
57 };

75 /// Reader smart pointer for rcu::Variable<T>. You may use operator*() or
76 /// operator->() to do something with the stored value. Once created,
77 /// ReadablePtr references the same immutable value: if Variable's value is
78 /// changed during ReadablePtr lifetime, it will not affect value referenced by
79 /// ReadablePtr.
80 template <typename T>
81 class USERVER_NODISCARD ReadablePtr final {
82  public:
83  explicit ReadablePtr(const Variable<T>& ptr)
84  : hp_record_(&ptr.MakeHazardPointer()) {
85  // This cycle guarantees that at the end of it both t_ptr_ and
86  // hp_record_->ptr will both be set to
87  // 1. something meaningful
88  // 2. and that this meaningful value was not removed between assigning to
89  // t_ptr_ and storing it in a hazard pointer
90  do {
91  t_ptr_ = ptr.GetCurrent();
92 
93  hp_record_->ptr.store(t_ptr_);
94  } while (t_ptr_ != ptr.GetCurrent());
95  }
96 
97  ReadablePtr(ReadablePtr<T>&& other) noexcept
98  : t_ptr_(other.t_ptr_), hp_record_(other.hp_record_) {
99  other.t_ptr_ = nullptr;
100  }
101 
102  ReadablePtr& operator=(ReadablePtr<T>&& other) noexcept {
103  // What do we have here?
104  // 1. 'other' may point to the same variable - or to a different one.
105  // 2. therefore, its hazard pointer may belong to the same list,
106  // or to a different one.
107 
108  // We can't move hazard pointers between different lists - there is simply
109  // no way to do this.
110  // We also can't simply swap two pointers inside hazard pointers and leave
111  // it be, because if 'other' is from different Variable, then our
112  // ScanRetiredList won't find any more pointer to current T* and will
113  // destroy object, while 'other' is still holding a hazard pointer to it.
114  // The thing is, whole ReadablePtr is just a glorified holder for
115  // hazard pointer + t_ptr_ just so that we don't have to load() atomics
116  // every time.
117  // so, lets just take pointer to hazard pointer inside other
118  UASSERT_MSG(this != &other, "Self assignment to RCU variable");
119  if (this == &other) {
120  // it is an error to assign to self. Do nothing in this case
121  return *this;
122  }
123 
124  // Get rid of our current hp_record_
125  if (t_ptr_) {
126  hp_record_->Release();
127  }
128  // After that moment, the content of our hp_record_ can't be used -
129  // no more hp_record_->xyz calls, because it is probably already reused in
130  // some other ReadablePtr. Also, don't call t_ptr_, it is probably already
131  // freed. Just take values from 'other'.
132  hp_record_ = other.hp_record_;
133  t_ptr_ = other.t_ptr_;
134 
135  // Now, it won't do us any good if there were two glorified things having
136  // pointer to same hp_record_. Kill the other one.
137  other.t_ptr_ = nullptr;
138  // We don't need to clean other.hp_record_, because other.t_ptr_ acts
139  // like a guard to it. As long as other.t_ptr_ is nullptr, nobody will
140  // use other.hp_record_
141 
142  return *this;
143  }
144 
145  ReadablePtr(const ReadablePtr<T>& other)
146  : ReadablePtr(other.hp_record_->owner) {}
147 
148  ReadablePtr& operator=(const ReadablePtr<T>& other) {
149  if (this != &other) *this = ReadablePtr<T>{other};
150  return *this;
151  }
152 
153  ~ReadablePtr() {
154  if (!t_ptr_) return;
155  UASSERT(hp_record_ != nullptr);
156  hp_record_->Release();
157  }
158 
159  const T* Get() const& {
160  UASSERT(t_ptr_);
161  return t_ptr_;
162  }

164  const T* Get() && { return GetOnRvalue(); }
165 
166  const T* operator->() const& { return Get(); }
167  const T* operator->() && { return GetOnRvalue(); }
168 
169  const T& operator*() const& { return *Get(); }
170  const T& operator*() && { return *GetOnRvalue(); }

172  private:
173  const T* GetOnRvalue() {
174  static_assert(!sizeof(T),
175  "Don't use temporary ReadablePtr, store it to a variable");
176  std::abort();
177  }
178 
179  // This is a pointer to actual data. If it is null, then we treat it as
180  // an indicator that this ReadablePtr is cleared and won't call
181  // any logic associated with hp_record_
182  T* t_ptr_;
183  // Our hazard pointer. It can be nullptr in some circumstances.
184  // Invariant is this: if t_ptr_ is not nullptr, then hp_record_ is also
185  // not nullptr and points to hazard pointer containing same T*.
186  // Thus, if t_ptr_ is nullptr, then hp_record_ is undefined.
187  impl::HazardPointerRecord<T>* hp_record_;
188 };

190 /// Smart pointer for rcu::Variable<T> for changing RCU value. It stores a
191 /// reference to a to-be-changed value and allows one to mutate the value (e.g.
192 /// add items to std::unordered_map). Changed value is not visible to readers
193 /// until explicit store by Commit. Only a single writer may own a WritablePtr
194 /// associated with the same Variable, so WritablePtr creates a critical
195 /// section. This critical section doesn't affect readers, so a slow writer
196 /// doesn't block readers.
197 /// @note you may not pass WritablePtr between coroutines as it owns
198 /// engine::Mutex, which must be unlocked in the same coroutine that was used to
199 /// lock the mutex.
200 template <typename T>
201 class USERVER_NODISCARD WritablePtr final {
202  public:
203  /// For internal use only. Use `var.StartWrite()` instead
204  explicit WritablePtr(Variable<T>& var)
205  : var_(var),
206  lock_(var.mutex_),
207  ptr_(std::make_unique<T>(*var_.GetCurrent())) {
208  LOG_TRACE() << "Start writing ptr=" << ptr_.get();
209  }
210 
211  /// For internal use only. Use `var.Emplace(args...)` instead
212  template <typename... Args>
213  WritablePtr(Variable<T>& var, std::in_place_t, Args&&... initial_value_args)
214  : var_(var),
215  lock_(var.mutex_),
216  ptr_(std::make_unique<T>(std::forward<Args>(initial_value_args)...)) {
217  LOG_TRACE() << "Start writing ptr=" << ptr_.get()
218  << " with custom initial value";
219  }
220 
221  WritablePtr(WritablePtr<T>&& other) noexcept
222  : var_(other.var_),
223  lock_(std::move(other.lock_)),
224  ptr_(std::move(other.ptr_)) {
225  LOG_TRACE() << "Continue writing ptr=" << ptr_.get();
226  }
227 
228  ~WritablePtr() {
229  if (ptr_) {
230  LOG_TRACE() << "Stop writing ptr=" << ptr_.get();
231  }
232  }

234  /// Store the changed value in Variable. After Commit() the value becomes
235  /// visible to new readers (IOW, Variable::Read() returns ReadablePtr
236  /// referencing the stored value, not an old value).
237  void Commit() {
238  UASSERT(ptr_ != nullptr);
239  LOG_TRACE() << "Committing ptr=" << ptr_.get();
240 
241  std::unique_ptr<T> old_ptr(var_.current_.exchange(ptr_.release()));
242  var_.Retire(std::move(old_ptr), lock_);
243  lock_.unlock();
244  }

247  UASSERT(ptr_);
248  return ptr_.get();
249  }

257  T& operator*() && { return *GetOnRvalue(); }

260  T* GetOnRvalue() {
261  static_assert(!sizeof(T),
262  "Don't use temporary WritablePtr, store it to a variable");
263  std::abort();
264  }
265 
266  Variable<T>& var_;
267  std::unique_lock<engine::Mutex> lock_;
268  std::unique_ptr<T> ptr_;
269 };

275 /// @ingroup userver_concurrency userver_containers
276 ///
277 /// @brief Read-Copy-Update variable
278 ///
279 /// @see Based on ideas from
280 /// http://www.drdobbs.com/lock-free-data-structures-with-hazard-po/184401890
281 /// with modified API.
282 ///
283 /// A variable with MT-access pattern "very often reads, seldom writes". It is
284 /// specially optimized for reads. On read, one obtains a ReaderPtr<T> from it
285 /// and uses the obtained value as long as it wants to. On write, one obtains a
286 /// WritablePtr<T> with a copy of the last version of the value, makes some
287 /// changes to it, and commits the result to update current variable value (does
288 /// Read-Copy-Update). Old version of the value is not freed on update, it will
289 /// be eventually freed when a subsequent writer identifies that nobody works
290 /// with this version.
291 ///
292 /// @note There is no way to create a "null" `Variable`.
293 ///
294 /// ## Example usage:
295 ///
296 /// @snippet rcu/rcu_test.cpp Sample rcu::Variable usage
297 ///
298 /// @see @ref md_en_userver_synchronization
299 template <typename T>
300 class Variable final {
301  public:
302  /// Create a new `Variable` with an in-place constructed initial value.
303  /// Asynchronous destruction is enabled by default.
304  /// @param initial_value_args arguments passed to the constructor of the
305  /// initial value
306  template <typename... Args>
307  Variable(Args&&... initial_value_args)
308  : destruction_type_((std::is_trivially_destructible_v<T> ||
309  std::is_same_v<T, std::string>)
310  ? DestructionType::kSync
311  : DestructionType::kAsync),
312  epoch_(impl::GetNextEpoch()),
313  current_(new T(std::forward<Args>(initial_value_args)...)) {}

315  /// Create a new `Variable` with an in-place constructed initial value.

317  /// be performed asynchronously

319  /// initial value

395  impl::HazardPointerRecord<T>* MakeHazardPointerCached() const {
396  auto& cache = impl::cache<T>;
397  auto* hp = cache.hp;
398  T* ptr = nullptr;
399  if (hp && cache.variable == this && cache.variable_epoch == epoch_) {
400  if (hp->ptr.load() == nullptr &&
401  hp->ptr.compare_exchange_strong(

410  impl::HazardPointerRecord<T>* MakeHazardPointerFast() const {
411  // Look for any hazard pointer with nullptr data ptr.
412  // Mark it with kUsed (to reserve it for ourselves) and return it.
413  auto* hp = hp_record_head_.load();
414  while (hp) {
415  T* t_ptr = nullptr;
416  if (hp->ptr.load() == nullptr &&
417  hp->ptr.compare_exchange_strong(
418  t_ptr, impl::HazardPointerRecord<T>::kUsed)) {
419  return hp;
420  }

427  impl::HazardPointerRecord<T>& MakeHazardPointer() const {
428  auto* hp = MakeHazardPointerCached();
429  if (!hp) {
430  hp = MakeHazardPointerFast();
431  // all buckets are full, create a new one
432  if (!hp) hp = MakeHazardPointerSlow();

443  impl::HazardPointerRecord<T>* MakeHazardPointerSlow() const {
444  // allocate new pointer, and add it to the list (atomically)
445  auto hp = new impl::HazardPointerRecord<T>(*this);
446  impl::HazardPointerRecord<T>* old_hp;
447  do {
448  old_hp = hp_record_head_.load();

454  void Retire(std::unique_ptr<T> old_ptr,
455  std::unique_lock<engine::Mutex>& lock) {
456  LOG_TRACE() << "Retiring ptr=" << old_ptr.get();
457  auto hazard_ptrs = CollectHazardPtrs(lock);
458 
459  if (hazard_ptrs.count(old_ptr.get()) > 0) {
460  // old_ptr is being used now, we may not delete it, delay deletion
461  LOG_TRACE() << "Not retire, still used ptr=" << old_ptr.get();
462  retire_list_head_.push_back(std::move(old_ptr));
463  } else {
464  LOG_TRACE() << "Retire, not used ptr=" << old_ptr.get();
465  DeleteAsync(std::move(old_ptr));
466  }
467 
468  ScanRetiredList(hazard_ptrs);
469  }

471  // Scan retired list and for every object that has no more hazard_ptrs
472  // pointing at it, destroy it (asynchronously)
473  void ScanRetiredList(const std::unordered_set<T*>& hazard_ptrs) {
474  for (auto rit = retire_list_head_.begin();
475  rit != retire_list_head_.end();) {
476  auto current = rit++;
477  if (hazard_ptrs.count(current->get()) == 0) {
478  // *current is not used by anyone, may delete it
479  DeleteAsync(std::move(*current));
480  retire_list_head_.erase(current);
481  }
482  }
483  }

485  // Returns all T*, that have hazard ptr pointing at them. Occasionally nullptr
486  // might be in result as well.
487  std::unordered_set<T*> CollectHazardPtrs(std::unique_lock<engine::Mutex>&) {
488  std::unordered_set<T*> hazard_ptrs;
489 
490  // Learn all currently used hazard pointers
491  for (auto* hp = hp_record_head_.load(); hp; hp = hp->next) {
492  hazard_ptrs.insert(hp->ptr.load());
493  }
494  return hazard_ptrs;
495  }

497  void DeleteAsync(std::unique_ptr<T> ptr) {
498  switch (destruction_type_) {
499  case DestructionType::kSync:
500  ptr.reset();
501  break;
502  case DestructionType::kAsync:
503  engine::CriticalAsyncNoSpan([ptr = std::move(ptr),
504  token = wait_token_storage_
505  .GetToken()]() mutable {
506  // Make sure *ptr is deleted before token is destroyed
507  ptr.reset();
508  }).Detach();
509  break;
510  }
511  }
