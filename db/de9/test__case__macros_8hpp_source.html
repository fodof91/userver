88 template <typename UserFixture>
89 class TestLauncherParametric
90  : public WithStaticMethods<
91  ::testing::TestWithParam<typename UserFixture::ParamType>,
92  UserFixture> {
93  public:
94  // Called from the UTEST_P macro
95  template <typename EnrichedTest>
96  static void RunTest(std::size_t thread_count) {
97  using ParamType = typename UserFixture::ParamType;
98  const auto& parameter = ::testing::TestWithParam<ParamType>::GetParam();
99 
100  // It seems impossible to seamlessly proxy 'ParamType' from the launcher to
101  // the enriched fixture without using gtest internals.
102  auto factory = std::make_unique<
103  testing::internal::ParameterizedTestFactory<EnrichedTest>>(parameter);
104 
105  utest::impl::DoRunTest(thread_count, [&] {
106  return std::unique_ptr<EnrichedTest>{
107  dynamic_cast<EnrichedTest*>(factory->CreateTest())};
108  });
109  }
110 };
