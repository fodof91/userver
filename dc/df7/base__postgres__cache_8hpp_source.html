519 template <typename PostgreCachePolicy>
520 void PostgreCache<PostgreCachePolicy>::Update(
521  cache::UpdateType type,
522  const std::chrono::system_clock::time_point& last_update,
523  const std::chrono::system_clock::time_point& /*now*/,
524  cache::UpdateStatisticsScope& stats_scope) {
525  namespace pg = storages::postgres;
526  if constexpr (!kIncrementalUpdates) {
527  type = cache::UpdateType::kFull;
528  }
529  const auto query =
530  (type == cache::UpdateType::kFull) ? GetAllQuery() : GetDeltaQuery();
531  const std::chrono::milliseconds timeout = (type == cache::UpdateType::kFull)
532  ? full_update_timeout_
533  : incremental_update_timeout_;
534 
535  // COPY current cached data
536  auto scope = tracing::Span::CurrentSpan().CreateScopeTime(
537  std::string{pg_cache::detail::kCopyStage});
538  auto data_cache = GetDataSnapshot(type, scope);
539  [[maybe_unused]] const auto old_size = data_cache->size();
540 
541  scope.Reset(std::string{pg_cache::detail::kFetchStage});
542 
543  size_t changes = 0;
544  // Iterate clusters
545  for (auto cluster : clusters_) {
546  if (chunk_size_ > 0) {
547  auto trx = cluster->Begin(
548  kClusterHostTypeFlags, pg::Transaction::RO,
549  pg::CommandControl{timeout, pg_cache::detail::kStatementTimeoutOff});
550  auto portal =
551  trx.MakePortal(query, GetLastUpdated(last_update, *data_cache));
552  while (portal) {
553  scope.Reset(std::string{pg_cache::detail::kFetchStage});
554  auto res = portal.Fetch(chunk_size_);
555  stats_scope.IncreaseDocumentsReadCount(res.Size());
556 
557  scope.Reset(std::string{pg_cache::detail::kParseStage});
558  CacheResults(res, data_cache, stats_scope, scope);
559  changes += res.Size();
560  }
561  trx.Commit();
562  } else {
563  bool has_parameter = query.Statement().find('$') != std::string::npos;
564  auto res = has_parameter
565  ? cluster->Execute(
566  kClusterHostTypeFlags,
567  pg::CommandControl{
568  timeout, pg_cache::detail::kStatementTimeoutOff},
569  query, GetLastUpdated(last_update, *data_cache))
570  : cluster->Execute(
571  kClusterHostTypeFlags,
572  pg::CommandControl{
573  timeout, pg_cache::detail::kStatementTimeoutOff},
574  query);
575  stats_scope.IncreaseDocumentsReadCount(res.Size());
576 
577  scope.Reset(std::string{pg_cache::detail::kParseStage});
578  CacheResults(res, data_cache, stats_scope, scope);
579  changes += res.Size();
580  }
581  }
582 
583  scope.Reset();
584 
585  if constexpr (pg_cache::detail::kIsContainerCopiedByElement<DataType>) {
586  if (old_size > 0) {
587  const auto elapsed_copy =
588  scope.ElapsedTotal(std::string{pg_cache::detail::kCopyStage});
589  if (elapsed_copy > pg_cache::detail::kCpuRelaxThreshold) {
590  cpu_relax_iterations_copy_ = static_cast<std::size_t>(
591  static_cast<double>(old_size) /
592  (elapsed_copy / pg_cache::detail::kCpuRelaxInterval));
593  LOG_TRACE() << "Elapsed time for copying " << kName << " "
594  << elapsed_copy.count() << " for " << changes
595  << " data items is over threshold. Will relax CPU every "
596  << cpu_relax_iterations_parse_ << " iterations";
597  }
598  }
599  }
600 
601  if (changes > 0) {
602  const auto elapsed_parse =
603  scope.ElapsedTotal(std::string{pg_cache::detail::kParseStage});
604  if (elapsed_parse > pg_cache::detail::kCpuRelaxThreshold) {
605  cpu_relax_iterations_parse_ = static_cast<std::size_t>(
606  static_cast<double>(changes) /
607  (elapsed_parse / pg_cache::detail::kCpuRelaxInterval));
608  LOG_TRACE() << "Elapsed time for parsing " << kName << " "
609  << elapsed_parse.count() << " for " << changes
610  << " data items is over threshold. Will relax CPU every "
611  << cpu_relax_iterations_parse_ << " iterations";
612  }
613  }
614  if (changes > 0 || type == cache::UpdateType::kFull) {
615  // Set current cache
616  stats_scope.Finish(data_cache->size());
617  pg_cache::detail::OnWritesDone(*data_cache);
618  this->Set(std::move(data_cache));
619  } else {
620  stats_scope.FinishNoChanges();
621  }
622 }

635  for (auto p = values.begin(); p != values.end(); ++p) {
