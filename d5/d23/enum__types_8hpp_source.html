22 /// @page pg_enum uPg: Mapping a C++ enum to PostgreSQL enum type.
23 ///
24 /// A C++ enumeration can be mapped to a PostgreSQL enum type by providing a
25 /// list of PostgreSQL literals mapped to the enumeration values.
26 ///
27 /// For example, if we have a C++ enumeration declared as follows:
28 /// @snippet storages/postgres/tests/enums_pgtest.cpp C++ enum type
29 /// and a PostgreSQL enum type:
30 /// @code
31 /// create type __pgtest.rainbow as enum (
32 /// 'red', 'orange', 'yellow', 'green', 'cyan', 'blue', 'violet'
33 /// )
34 /// @endcode
35 /// all we need to do to declare the mapping between the C++ type and PosgtreSQL
36 /// enumeration is provide a specialisation of CppToUserPg template.
37 /// The difference from mapping a C++ type to other user PosgreSQL types, for an
38 /// enumeration we need to provide a list of enumerators with corresponding
39 /// literals.
40 ///
41 /// @warning The type mapping specialisation **must** be accessible at the
42 /// points where parsing/formatting of the C++ type is instantiated. The
43 /// header where the C++ type is declared is an appropriate place to do it.
44 ///
45 /// @snippet storages/postgres/tests/enums_pgtest.cpp C++ to Pg mapping
46 ///
47 /// The specialisation of CppToUserPg derives from EnumMappingBase for it to
48 /// provide type aliases for EnumeratorList and EnumType. EnumType is an alias
49 /// to the enumeration type for convenience of declaring pairs, as the
50 /// enumeration can have a long qualified name.
51 ///
52

122  static MapsPair MapLiterals() {
123  MapsPair maps;
124  for (const auto& enumerator : enumerators) {
125  maps.first.insert(
126  std::make_pair(enumerator.enumerator, enumerator.literal));
127  maps.second.insert(
128  std::make_pair(enumerator.literal, enumerator.enumerator));
129  }
130  return maps;
131  }

133  static auto maps_ = MapLiterals();
134  return maps_;
135  }
136  static const EnumToString& EnumeratorMap() { return GetMaps().first; }
137  static const StringToEnum& LiteralMap() { return GetMaps().second; }
138 
139  public:
140  static constexpr const auto& enumerators = MappingType::enumerators;
141  static constexpr std::size_t size = std::size(enumerators);
142  static EnumType GetEnumerator(StringType literal) {
143  static const auto& map = LiteralMap();
144  if (auto f = map.find(literal); f != map.end()) {
145  return f->second;
146  }
147  throw InvalidEnumerationLiteral{
