7 /// @page pg_range_types uPg: Range types
8 ///
9 /// PostgreSQL provides a facility to represent intervals of values. They can be
10 /// bounded (have both ends), e.g [0, 1], [2, 10), unbounded (at least one end
11 /// is infinity or absent), e.g. (-∞, +∞), and empty.
12 ///
13 /// The range that can be unbounded is modelled by storages::postgres::Range<>
14 /// template, which provides means of constructing any possible combination of
15 /// interval ends. It is very versatile, but not very convenient in most cases.
16 /// storages::postgres::BoundedRange<> template is an utility that works only
17 /// with bounded ranges.
18 ///
19 /// @par PostgreSQL Built-in Range Datatypes
20 ///
21 /// Some of PostgreSQL built-in range datatypes
22 /// (https://www.postgresql.org/docs/current/rangetypes.html#RANGETYPES-BUILTIN)
23 /// are supported by the library, please see @ref pg_types
24 ///
25 /// @par User Range Types
26 ///
27 /// A user can define custom range types
28 /// (https://www.postgresql.org/docs/current/rangetypes.html#RANGETYPES-DEFINING)
29 /// and they can be mapped to C++ counterparts, e.g.
30 ///
31 /// @code
32 /// CREATE TYPE my_range AS RANGE (
33 /// subtype = my_type
34 /// );
35 /// @endcode
36 ///
37 /// and declare a mapping from C++ range to this type just as for any other user
38 /// type:
39 ///
40 /// @code
41 /// template<>
42 /// struct CppToUserPg<Range<my_type>> {
43 /// static constexpr DBTypeName postgres_name = "my_type";
44 /// };
45 /// @endcode
46 ///
47 /// Please note that `my_type` must be comparable both in Postgres and in C++
48 ///
49 /// See @ref pg_user_types for more info
50 ///
51 /// @par Time Range and Other Widely Used Types
52 ///
53 /// If you need a range of PostgreSQL `float` type or `time` type (actually any
54 /// type mapped to C++ type that is highly likely used by other developers),
55 /// please DO NOT specialize mapping for `Range<float>`, `Range<double>` or
56 /// `Range<TimeOfDay<seconds>>`. Please declare a strong typedef for your range
57 /// or bounded range and map it to your postgres range type.
58 ///
59 /// Here is an example how to define a strong typedef for a range of TimeOfDay
60 /// and map it to postgres user range type.
61 ///
62 /// @snippet storages/postgres/tests/user_types_pgtest.cpp Time range
63 ///
64 /// @snippet storages/postgres/tests/user_types_pgtest.cpp Range type mapping
65 ///

359 template <typename T>
360 struct RangeBinaryParser : BufferParserBase<Range<T>> {
361  using BaseType = BufferParserBase<Range<T>>;
362  using ValueType = typename BaseType::ValueType;
363  using ElementType = T;
364  using ElementParser = typename traits::IO<ElementType>::ParserType;
365 
366  static constexpr BufferCategory element_buffer_category =
367  traits::kParserBufferCategory<ElementParser>;
368 
369  using BaseType::BaseType;
370 
371  void operator()(FieldBuffer buffer, const TypeBufferCategory& categories) {
372  char wire_range_flags{0};
373 
374  buffer.Read(wire_range_flags, BufferCategory::kPlainBuffer);
375  RangeFlags range_flags(static_cast<RangeFlag>(wire_range_flags));
376 
377  ValueType wire_value;
378  if (range_flags != RangeFlag::kEmpty) {
379  RangeBounds bounds = RangeBound::kNone;
380  typename ValueType::OptionalValue lower;
381  typename ValueType::OptionalValue upper;
382  if (HasLowerBound(range_flags)) {
383  if (range_flags & RangeFlag::kLowerBoundInclusive) {
384  bounds |= RangeBound::kLower;
385  }
386  T tmp;
387  buffer.ReadRaw(tmp, categories, element_buffer_category);
388  lower = tmp;
389  }
390  if (HasUpperBound(range_flags)) {
391  if (range_flags & RangeFlag::kUpperBoundInclusive) {
392  bounds |= RangeBound::kUpper;
393  }
394  T tmp;
395  buffer.ReadRaw(tmp, categories, element_buffer_category);
396  upper = tmp;
397  }
398  wire_value = ValueType{lower, upper, bounds};
399  }
400  this->value = wire_value;
401  }
402 };

404 template <typename T>
405 struct RangeBinaryFormatter : BufferFormatterBase<Range<T>> {
406  using BaseType = BufferFormatterBase<Range<T>>;
407 
408  using BaseType::BaseType;
409 
410  template <typename Buffer>
411  void operator()(const UserTypes& types, Buffer& buffer) const {
412  RangeFlags range_flags;
413  if (this->value.Empty()) {
414  range_flags |= RangeFlag::kEmpty;
415  } else {
416  // Mark lower/upper bound
417  if (!this->value.HasLowerBound()) {
418  range_flags |= RangeFlag::kLowerBoundInfinity;
419  } else if (this->value.IsLowerBoundIncluded()) {
420  range_flags |= RangeFlag::kLowerBoundInclusive;
421  }
422  if (!this->value.HasUpperBound()) {
423  range_flags |= RangeFlag::kUpperBoundInfinity;
424  } else if (this->value.IsUpperBoundIncluded()) {
425  range_flags |= RangeFlag::kUpperBoundInclusive;
426  }
427  }
428  char wire_range_flags = static_cast<char>(range_flags.GetValue());
429  io::WriteBuffer(types, buffer, wire_range_flags);
430  if (!this->value.Empty()) {
431  // Write lower/upper bounds
432  if (this->value.HasLowerBound()) {
433  io::WriteRawBinary(types, buffer, this->value.GetLowerBound());
434  }
435  if (this->value.HasUpperBound()) {
436  io::WriteRawBinary(types, buffer, this->value.GetUpperBound());
437  }
438  }
439  }

589 template <typename T>
590 bool Range<T>::RangeData::BoundEqual(const RangeData& rhs,
591  RangeBounds side) const {
592  bool has_bound = HasBound(side);
593  if (has_bound != rhs.HasBound(side)) {
594  return false;
595  }
596  if (!has_bound) { // both are unbounded
597  return true;
598  }
599  const auto& lval = GetBoundValue(side);
600  const auto& rval = rhs.GetBoundValue(side);
601  if ((bounds & side) == (rhs.bounds & side)) {
602  // same include
603  return lval == rval;
604  }
605  if constexpr (kIsDiscreteValue) {
606  T diff = (side == RangeBound::kLower ? 1 : -1);
607  if (IsBoundIncluded(side)) {
608  return lval == rval + diff;
609  } else {
610  return lval + diff == rval;
611  }
612  }
613  return false;
614 }
