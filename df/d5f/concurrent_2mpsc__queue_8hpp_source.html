161 template <typename T>
162 MpscQueue<T>::~MpscQueue() {
163  UASSERT(consumer_is_created_and_dead_ || !consumer_is_created_);
164  UASSERT(!producers_count_);
165  // Clear remaining items in queue. This will work for unique_ptr as well.
166  T value;
167  ConsumerToken temp_token{queue_};
168  while (PopNoblock(temp_token, value)) {
169  }
170 
171  if (consumer_is_created_and_dead_) {
172  remaining_capacity_.lock_shared_count(kSemaphoreUnlockValue);
173  }
174 
175  SetSoftMaxSize(kUnbounded);
176 }

194 template <typename T>
195 void MpscQueue<T>::SetSoftMaxSize(size_t max_size) {
196  max_size = std::min(max_size, kUnbounded);
197  const auto old_capacity = capacity_.exchange(max_size);
198  if (max_size > old_capacity) {
199  remaining_capacity_.unlock_shared_count(max_size - old_capacity);
200  } else if (max_size < old_capacity) {
201  remaining_capacity_.lock_shared_count(old_capacity - max_size);
202  }
203 }

244 template <typename T>
245 bool MpscQueue<T>::DoPush(ProducerToken& /*unused*/, T&& value) {
246  if (consumer_is_created_and_dead_) {
247  remaining_capacity_.unlock_shared();
248  return false;
249  }

258 template <typename T>
259 bool MpscQueue<T>::Pop(ConsumerToken& token, T& value,
260  engine::Deadline deadline) {
261  while (!DoPop(token, value)) {
262  if (producer_is_created_and_dead_ ||
263  !nonempty_event_.WaitForEventUntil(deadline)) {
264  // Producer might have pushed something in queue between .pop()
265  // and !producer_is_created_and_dead_ check. Check twice to avoid
266  // TOCTOU.
267  return DoPop(token, value);
268  }
269  }
270  return true;
271 }

278 template <typename T>
279 bool MpscQueue<T>::DoPop(ConsumerToken& /*unused*/, T& value) {
280  if (QueueHelper::Pop(queue_, value)) {
281  --size_;
282  remaining_capacity_.unlock_shared();
283  nonempty_event_.Reset();
284  return true;
285  }
286  return false;
287 }
